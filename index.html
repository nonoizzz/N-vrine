        // Typewriter effect for headline

        window.addEventListener('load', () => {

            const headline = document.querySelector('.main-headline');

            const originalText = headline.innerHTML;

            headline.innerHTML = '';

           

            let i = 0;

            const text = originalText.replace('<br>', '\n');

           

            const typeWriter = () => {

                if (i < text.length) {

                    if (text.charAt(i) === '\n') {

                        headline.innerHTML += '<br>';

                    } else {

                        headline.innerHTML += text.charAt(i);

                    }

                    i++;

                    setTimeout(typeWriter, 80);

                } else {

                    headline.innerHTML = originalText;

                }

            };

           

            setTimeout(typeWriter, 600);

        });
